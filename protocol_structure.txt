->General structure:
[field_name (byte size of field)]
header = [id (128)] [cmd(1)] [path_len(4)] [data_len(8)] [path (path_len)]
MSS = 200

-> First message:
if there is id:
[1] [id] [path_len] [path]
if there is no id yet:
[0] [path_len] [path]

-> NEW (file \ folder)
cmd = 0x00
sending:
    if new file:
        send header
        while file not empty : send(file.read(MSS))
    if new folder:
        send header
        for each file : new file
        for each folder : new folder recursively

-> DEL
cmd = 0x02
data_len = 0
sending:
    send header

-> MOVE:
cmd = 0x03
path = where moved from
data_len = 0
sending:
    send header:
        cmd = MOVF = 0x03
        path = moved from
    send where_to:
        cmd = MOVT = 0x04
        path = move to
